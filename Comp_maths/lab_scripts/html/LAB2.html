function [x,iterationCount]=JacobiMethod(A,b,x_0,tol)
    %Decomposing the metrix into its U, P, and D components
    L = tril(A,-1);
    U = tril(A',-1)';
    D = diag(A);
    D = diag(D);

    iterationCount = 0;
    x = x_0;
    xold = x_0;
    while true
        y = b - (L+U)*x;
        x = D\y;
        iterationCount = iterationCount + 1;
        %Getting and checking the relative error
        err = max(abs((x-xold)./x));
        if err<=tol
            break;
        end
        xold = x;
    end
end


function [x,iterationCount]= gaussSeidel(A,b,x_0,tol)
    %Decomposing the metrix into its U, P, and D components
    L = tril(A,-1);
    U = tril(A',-1)';
    D = diag(A);
    D = diag(D);

    iterationCount = 0;
    x = x_0;
    xold = x_0;
    while true
        y = b - U*x;
        x = (L+D)\y;
        iterationCount = iterationCount + 1;

        %Getting and checking the relative error
        err = max(abs((x-xold)./x));
        if err<=tol
            break;
        end
        xold = x;
    end
end
